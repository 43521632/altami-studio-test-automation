# --- Управление ВМ -----------------------------------------------------------
# ВАЖНО: libvirt-python собирается из исходников и требует системных заголовков.
# На Ubuntu сначала выполните:
#   sudo apt install -y virt-manager libvirt-daemon-system libvirt-dev pkg-config
# Альтернатива без сборки — системный пакет python3-libvirt
# (тогда venv нужно создавать с флагом --system-site-packages).
libvirt-python>=8.0.0

# --- Управление гостевым UI (QMP) --------------------------------------------
# Официальная библиотека QEMU: https://gitlab.com/qemu-project/python-qemu-qmp
# Даёт input-send-event (клавиатура + abs-мышь) и screendump — то, чего libvirt
# не умеет. Чистый Python, без обязательных зависимостей. Требует Python 3.8+.
# ВАЖНО: подключается ко ВТОРОМУ QMP-сокету ВМ, а не к монитору libvirt —
# libvirt держит свой монитор монопольно. См. README, раздел «Второй QMP-сокет».
qemu.qmp>=0.0.6
